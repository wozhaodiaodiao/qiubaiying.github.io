    vue-ssr 思路

    简单点说下出场人物吧，vue，koa，vue-server-renderer，webpack

    vue：我用vue-cli3创建的项目作为前端（这点埋下了个坑，后面会说）

    koa：我用koa创建一个服务端

    vue-server-renderer：vue服务端渲染的库

    webpack：分别进行配置使其三者融合好，而且还通过它的devServer作为一个服务器与node端进行了通信（这个后面会重点讲），然后分别在client和server的配置文件中使用vue-server-renderer的client-plugin和server-plugin进行插件注册，目的生成对应的清单文件（你可以理解为打包好的文件是json）二者都有默认名称分别是vue-ssr-client-manifest.json和vue-ssr-server-bundle.json

    流程

    1. 打包前端代码，然后执行打包命令生成vue-ssr-client-manifest.json，此时webpapck启动了devserver
    2. 使用webpack对服务端进行打包生成一个bundle，同时通过axios去获取前端打包好的vue-ssr-client-manifest.json,然后再读取模版文件server.template.ejs
    3. 然后创建vue-server-renderer（服务端渲染库）的实例，将bundle，vue-ssr-client-manifest.json，server.template.ejs作为参数传入进这个实例，然后生成一个renderer对象并调用renderToString函数传入一个context上下文对象作为插值生成一个appString，实际上就是与webpack打包的entry文件进行交互，通过context的url进行服务器端的路由设置，从而渲染运行。我拿个图吧。。。。

    webpack配置

    client配置

    const ClientPlugin = require('vue-server-renderer/client-plugin'),//生成客户端清单
    nodeExternals = require('webpack-node-externals');//忽略node_modules文件夹中的所有模块
module.exports = {
    devServer: {
        proxy: 'http://localhost:8080'//通过这个服务地址进行前端打包文件的获取
    },
    outputDir: 'dist/',
    configureWebpack: (config) => {
        return ({
            entry: `./src/entry/client`,
            output: {
                filename: 'js/[name].js',
                chunkFilename: 'js/[name].js',
                libraryTarget:  undefined
            },
            target:  'web',
            externals:  undefined,
            plugins: [
               new ClientPlugin(),//根据环境调用不同的渲染插件。
            ]
        })
    },
    chainWebpack: config => {
        config.resolve.alias
            .set('vue$', 'vue/dist/vue.esm.js')
        config.module
            .rule('vue')
            .use('vue-loader')
            .tap(options => {
                options.optimizeSSR = false;
                return options;
            });
        config.module
            .rule('images')
            .use('url-loader')
            .tap(options => {
                options = {
                    limit: 1024,
                    fallback: 'file-loader?name=img/[path][name].[ext]'
                }
                return options;
            });
    }
}

    server配置

    const ServerPlugin = require('vue-server-renderer/server-plugin'),//生成服务端清单
  nodeExternals = require('webpack-node-externals');//忽略node_modules文件夹中的所有模块
// const ExtractPlugin = require('extract-text-webpack-plugin')
const utils = require('util')
const path = require('path')
// const merge = require('webpack-merge')
// const baseConfig = require('./webpack.config.base');
const createVueLoaderOptions = require('./vue-loader.config')
const VueLoaderPlugin = require('vue-loader/lib/plugin');
const isDev = process.env.NODE_ENV === 'development'
module.exports = {
  target: 'node',//1，首先指定node平台，这个是要在服务器端进行配置的
  entry: `./src/entry/server`,//2. 指定入口文件
  devtool: 'source-map',//3. 开启sourcemap
  output: {
    libraryTarget: 'commonjs2',
    filename: 'server-entry.js',
    path: path.join(__dirname, '../distDev')
  },
  externals: Object.keys(require('../package.json').dependencies),//一些库在前端都已经打包好了，服务端做下排除
  plugins: [
    new ServerPlugin(),//真多服务器端点用服务器端插件
    new VueLoaderPlugin(),//不加不行。。
  ],
  module: {
    rules: [
      {
        test: /\.(vue|js|jsx)$/,
        loader: 'eslint-loader',
        exclude: /node_modules/,
        enforce: 'pre'
      },
      {
        test: /\.vue$/,
        loader: 'vue-loader',
        options: createVueLoaderOptions(isDev)
      },
      {
        test: /\.jsx$/,
        loader: 'babel-loader'
      },
      {
        test: /\.js$/,
        loader: 'babel-loader',
        exclude: /node_modules/
      },
      {
        test: /\.(gif|jpg|jpeg|png|svg)$/,
        use: [
          {
            loader: 'url-loader',
            options: {
              limit: 1024,
              name: 'resources/[path][name].[hash:8].[ext]'
            }
          }
        ]
      }
    ]
  }
}
